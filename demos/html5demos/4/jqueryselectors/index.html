Enter Your Own Selectors ( example: li:nth-child(2) )

Basics

$('code')元素选择器
$('#myid')ID选择器
$('.myclass')类选择器
$('*')选择全部
$('code, #myid, .myclass')多选择

Hierarchy

$('div code')子类选择器
$('li > ul')直接子类选择器
$('strong + em')直接兄弟选择器
$('strong ~ em')兄弟选择器

Basic Filters

$('li:first')第一个元素
$('li:last')最后一个元素
$('li:not(li:first)')不是第一个元素
$('li:even')偶数
$('li:odd')奇数
$('li:eq(1)')下标为1
$('li:gt(2)')下标大于2
$('li:lt(2)')下标小于2
$(':header')h标签
$(':animated')不清楚

Content Filters

$('li:contains(second-level)')2级
$(':empty')没有儿子
$('li:has(a)')有a标签
$('p:parent')有父标签

Visibility Filters

$(':hidden')隐藏
$(':visible')可见

Attribute Filters

$('li[class]')含class属性
$('a[rel="self"]')rel属性值为self
$('a[rel!="nofollow"]')rel属性值不为nofollow
$('[class^="my"]')class属性值开头为my
$('a[title$="blog"]')title属性值结尾为blog
$('a[href*="zip"]')href属性值含zip
$('a[rel][href][title$="blog"]')多属性包含

Child Filters

$('li:nth-child(even)')第偶数孩纸
$('li:nth-child(odd)')第奇数孩纸
$('li:nth-child(2)')第2孩纸
$('li:nth-child(2n)')第2n孩纸
$('li:first-child')第一孩纸
$('li:last-child')最后孩纸
$('code:only-child')独生子

Forms

$(':input')input
$(':text')input类型为text
$(':password')input类型为password
$(':radio')input类型为radio
$(':checkbox')input类型为checkbox
$(':submit')input类型为submit
$(':image')input类型为image
$(':reset')input类型为reset
$(':button')input类型为button
$(':file')input类型为file
$(':hidden')input类型为hidden

Form Filters

$('input:enabled')input enabled
$(':disabled')disabled
$(':checked')(checkbox)checked
$(':selected')(option)selected
