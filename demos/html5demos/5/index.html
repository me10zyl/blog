jQuery each() attr() data()方法 

    //jQuery的遍历
    $("#each").click(function () {
    $("div").each(function () {
    alert($(this).html());
    });
    })
    //jQuery取属性
    $("#alertAttr").click(function () {
    alert($("#test").attr("abcd"));
    }
    )
    //jQuery更改属性
    $("#changeAttr").click(
    function () {
    $("#test").attr("abcd", "我改了abcd的属性");
    $("#mydiv input").attr("type", "button")
    }
    )
    //jQuery删除属性
    $("#removeAttr").click(
    function () {
    $("#test").removeAttr("abcd");
    }
    )
    //jQueryData属性
    $("#dataAttr").click(
    function () {
    alert($("#test").data("abcd"));
    }
    )
