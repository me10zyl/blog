游戏 

    <button id="btn_play">play game</button>
    <button id="btn_pause">pause game</button>
    <script>
        var animID;
        function play()
        {
            animID =  setInterval("console.log('play')",300);
        }
        function pause()
        {
            if(animID)/*如果animID不是undefined*/
                clearInterval(animID);
        }
        btn_play = document.getElementById('btn_play');
        btn_pause = document.getElementById('btn_pause');
        btn_play.onclick = play;
        btn_pause.onclick = pause;
    </script>
